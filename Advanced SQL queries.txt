Advanced SQL queries

SELECT
    attempts.unique_id,
    IF(attempts.unique_id != id.unique_id, 1, 0) AS is_different
FROM
    attempts
LEFT JOIN
    id
ON
    id.unique_id = attempts.unique_id

SELECT id.*, attempts.* FROM id, attempts WHERE id.unique_id = attempts.unique_id;
SELECT t1.*, t2.* FROM t1, t2 WHERE t1.i1 = t2.i2;

SELECT id.`first_name`, id.`last_name`, id.`student_number`, attempts.`created`, attempts.`attempt`, attempts.`score` FROM id, attempts WHERE id.unique_id = attempts.unique_id;
